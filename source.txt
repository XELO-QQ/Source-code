-- Insane Instance Saver v2
-- Optimized for low lag, backup-safe saves, autosave, restore, and a clean UI.

local HttpService = game:GetService("HttpService")
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local StarterGui = game:GetService("StarterGui")

local function safeCall(fn, ...)
	local ok, result = pcall(fn, ...)
	if ok then
		return true, result
	end
	return false, result
end

local function ensureFolder(path)
	if not path or path == "" then
		return
	end
	local ok = pcall(function()
		if not isfolder(path) then
			makefolder(path)
		end
	end)
	if not ok and string.find(path, "/") then
		local parent = path:match("^(.*)/[^/]+$")
		if parent and parent ~= path then
			ensureFolder(parent)
			pcall(function()
				if not isfolder(path) then
					makefolder(path)
				end
			end)
		end
	end
end

local function writeFile(path, text)
	ensureFolder(path:match("^(.*)/[^/]+$"))
	pcall(function()
		writefile(path, text)
	end)
end

local function readFile(path)
	local ok, data = pcall(function()
		return readfile(path)
	end)
	if ok then
		return data
	end
	return nil
end

local function fileExists(path)
	local ok, _ = pcall(function()
		return readfile(path)
	end)
	return ok
end

local function normalizeValue(value)
	local valueType = typeof(value)
	if valueType == "nil" then
		return nil
	elseif valueType == "boolean" or valueType == "number" or valueType == "string" then
		return value
	elseif valueType == "EnumItem" then
		return {__type = "EnumItem", enumType = value.EnumType.Name, value = value.Name}
	elseif valueType == "Vector2" then
		return {__type = "Vector2", x = value.X, y = value.Y}
	elseif valueType == "Vector3" then
		return {__type = "Vector3", x = value.X, y = value.Y, z = value.Z}
	elseif valueType == "UDim" then
		return {__type = "UDim", scale = value.Scale, offset = value.Offset}
	elseif valueType == "UDim2" then
		return {__type = "UDim2", xScale = value.X.Scale, xOffset = value.X.Offset, yScale = value.Y.Scale, yOffset = value.Y.Offset}
	elseif valueType == "Color3" then
		return {__type = "Color3", r = value.R, g = value.G, b = value.B}
	elseif valueType == "BrickColor" then
		return {__type = "BrickColor", name = tostring(value.Name)}
	elseif valueType == "CFrame" then
		local x, y, z, R00, R01, R02, R10, R11, R12, R20, R21, R22 = value:GetComponents()
		return {__type = "CFrame", x = x, y = y, z = z, r00 = R00, r01 = R01, r02 = R02, r10 = R10, r11 = R11, r12 = R12, r20 = R20, r21 = R21, r22 = R22}
	elseif valueType == "ColorSequence" then
		return {__type = "ColorSequence", value = tostring(value)}
	elseif valueType == "NumberSequence" then
		return {__type = "NumberSequence", value = tostring(value)}
	elseif valueType == "Rect" then
		return {__type = "Rect", minX = value.Min.X, minY = value.Min.Y, maxX = value.Max.X, maxY = value.Max.Y}
	elseif valueType == "Ray" then
		return {__type = "Ray", origin = normalizeValue(value.Origin), direction = normalizeValue(value.Direction)}
	elseif valueType == "PhysicalProperties" then
		return {__type = "PhysicalProperties", density = value.Density, friction = value.Friction, elasticity = value.Elasticity, frictionWeight = value.FrictionWeight, elasticityWeight = value.ElasticityWeight}
	elseif valueType == "Region3" then
		return {__type = "Region3", min = normalizeValue(value.Min), max = normalizeValue(value.Max)}
	elseif valueType == "Region3int16" then
		return {__type = "Region3int16", min = normalizeValue(value.Min), max = normalizeValue(value.Max)}
	elseif valueType == "Instance" then
		return {__type = "InstanceRef", className = value.ClassName, fullName = value:GetFullName()}
	else
		return {__type = "String", value = tostring(value)}
	end
end

local function restoreValue(value)
	if type(value) ~= "table" then
		return value
	end
	local valueType = value.__type
	if not valueType then
		return value
	elseif valueType == "EnumItem" then
		return Enum[value.enumType][value.value]
	elseif valueType == "Vector2" then
		return Vector2.new(value.x, value.y)
	elseif valueType == "Vector3" then
		return Vector3.new(value.x, value.y, value.z)
	elseif valueType == "UDim" then
		return UDim.new(value.scale, value.offset)
	elseif valueType == "UDim2" then
		return UDim2.new(value.xScale, value.xOffset, value.yScale, value.yOffset)
	elseif valueType == "Color3" then
		return Color3.new(value.r, value.g, value.b)
	elseif valueType == "BrickColor" then
		return BrickColor.new(value.name)
	elseif valueType == "CFrame" then
		return CFrame.new(value.x, value.y, value.z, value.r00, value.r01, value.r02, value.r10, value.r11, value.r12, value.r20, value.r21, value.r22)
	elseif valueType == "String" then
		return value.value
	elseif valueType == "InstanceRef" then
		return nil
	else
		return value
	end
end

local function buildSignature(instance)
	local props = {}
	for _, propName in ipairs({"Name", "ClassName", "Archivable", "Parent"}) do
		props[#props + 1] = tostring(instance[propName])
	end
	return HttpService:GenerateGUID() .. "|" .. table.concat(props, "|") .. "|" .. #instance:GetChildren()
end

local function getSafeProperties(instance)
	local props = {}
	local list = instance:GetProperties()
	for _, propName in ipairs(list) do
		if propName ~= "Parent" and propName ~= "Archivable" then
			local ok, value = safeCall(function()
				return instance[propName]
			end)
			if ok then
				props[propName] = normalizeValue(value)
			end
		end
	end
	return props
end

local function collectInstanceData(instance, options)
	local data = {
		className = instance.ClassName,
		name = instance.Name,
		properties = getSafeProperties(instance),
		attributes = {},
		children = {},
	}

	if options.includeAttributes then
		for attrName, attrValue in pairs(instance:GetAttributes()) do
			data.attributes[attrName] = normalizeValue(attrValue)
		end
	end

	if options.includeChildren then
		local children = instance:GetChildren()
		for i = 1, #children do
			data.children[i] = collectInstanceData(children[i], options)
		end
	end

	return data
end

local function createInstanceFromData(data, parent)
	local instance = Instance.new(data.className)
	instance.Name = data.name or instance.Name

	for propName, propValue in pairs(data.properties or {}) do
		local ok = pcall(function()
			instance[propName] = restoreValue(propValue)
		end)
		if not ok then
			-- Skip unsupported props safely.
		end
	end

	for attrName, attrValue in pairs(data.attributes or {}) do
		instance:SetAttribute(attrName, restoreValue(attrValue))
	end

	for _, childData in ipairs(data.children or {}) do
		createInstanceFromData(childData, instance)
	end

	if parent then
		instance.Parent = parent
	end

	return instance
end

local function timestampLabel()
	return os.date("%Y-%m-%d_%H-%M-%S")
end

local Saver = {}
Saver.__index = Saver

function Saver.new(options)
	options = options or {}
	local self = setmetatable({}, Saver)
	self.options = {
		saveFolder = options.saveFolder or "InsaneSaver",
		backupFolder = options.backupFolder or "InsaneSaver/Backups",
		includeChildren = options.includeChildren ~= false,
		includeAttributes = options.includeAttributes ~= false,
		keepBackups = options.keepBackups or 8,
		autosaveInterval = options.autosaveInterval or 60,
		uiEnabled = options.uiEnabled ~= false,
	}
	self.session = {}
	self.autoSaveActive = false
	self.autoSaveThread = nil
	self.lastSignature = nil
	self.lastPath = nil
	self.lastRoot = nil
	return self
end

function Saver:Save(target, fileName, mode)
	mode = mode or "snapshot"
	local root = target or workspace
	local name = fileName or root.Name
	local folder = self.options.saveFolder
	local path = folder .. "/" .. name .. "_" .. mode .. ".json"
	local data = collectInstanceData(root, self.options)
	local encoded = HttpService:JSONEncode({
		meta = {
			version = 2,
			timestamp = timestampLabel(),
			mode = mode,
			className = root.ClassName,
			name = root.Name,
		},
		instance = data,
	})
	writeFile(path, encoded)
	self.lastPath = path
	self.lastRoot = root
	self.session[#self.session + 1] = {path = path, timestamp = timestampLabel(), root = root}

	local backupPath = self.options.backupFolder .. "/" .. name .. "_" .. timestampLabel() .. ".json"
	writeFile(backupPath, encoded)
	self:TrimBackups(name)
	return path
end

function Saver:TrimBackups(name)
	local folder = self.options.backupFolder
	ensureFolder(folder)
	local files = {}
	local ok, list = pcall(function()
		return listfiles(folder)
	end)
	if not ok then
		return
	end
	for _, filePath in ipairs(list) do
		if filePath:find(name .. "_") then
			files[#files + 1] = filePath
		end
	end
	table.sort(files, function(a, b)
		return a < b
	end)
	while #files > self.options.keepBackups do
		local oldPath = table.remove(files, 1)
		pcall(function()
			if isfile(oldPath) then
				delfile(oldPath)
			end
		end)
	end
end

function Saver:AutoSave(target, fileName, interval)
	interval = interval or self.options.autosaveInterval
	local root = target or workspace
	if self.autoSaveActive then
		self:StopAutoSave()
	end
	self.autoSaveActive = true
	self.autoSaveThread = task.spawn(function()
		while self.autoSaveActive do
			local sig = buildSignature(root)
			if sig ~= self.lastSignature then
				self:Save(root, fileName or root.Name, "autosave")
				self.lastSignature = sig
			end
			task.wait(interval)
		end
	end)
	return true
end

function Saver:StopAutoSave()
	self.autoSaveActive = false
	if self.autoSaveThread then
		pcall(task.cancel, self.autoSaveThread)
		self.autoSaveThread = nil
	end
end

function Saver:Restore(path, parent)
	local content = readFile(path)
	if not content then
		return nil
	end
	local decoded = HttpService:JSONDecode(content)
	local instanceData = decoded.instance
	local restored = createInstanceFromData(instanceData, parent)
	return restored
end

function Saver:ListSaves()
	local folder = self.options.saveFolder
	local out = {}
	local ok, list = pcall(function()
		return listfiles(folder)
	end)
	if ok then
		for _, filePath in ipairs(list) do
			if filePath:find(".json$") then
				out[#out + 1] = filePath
			end
		end
	end
	return out
end

function Saver:CreateUI()
	if not self.options.uiEnabled then
		return nil
	end
	local player = Players.LocalPlayer
	if not player then
		return nil
	end
	local screenGui = Instance.new("ScreenGui")
	screenGui.Name = "InsaneInstanceSaverUI"
	screenGui.ResetOnSpawn = false
	screenGui.Parent = player:WaitForChild("PlayerGui")

	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(0, 280, 0, 210)
	frame.Position = UDim2.new(0.02, 0, 0.1, 0)
	frame.BackgroundColor3 = Color3.fromRGB(16, 16, 24)
	frame.BorderSizePixel = 0
	frame.Parent = screenGui

	local title = Instance.new("TextLabel")
	title.Size = UDim2.new(1, 0, 0, 36)
	title.BackgroundTransparency = 1
	title.Text = "Insane Instance Saver"
	title.TextColor3 = Color3.fromRGB(255, 255, 255)
	title.TextScaled = true
	title.Parent = frame

	local function makeButton(name, text, callback)
		local button = Instance.new("TextButton")
		button.Size = UDim2.new(1, -20, 0, 34)
		button.Position = UDim2.new(0, 10, 0, 40 + (#frame:GetChildren() - 1) * 38)
		button.BackgroundColor3 = Color3.fromRGB(38, 38, 48)
		button.TextColor3 = Color3.fromRGB(255, 255, 255)
		button.Text = text
		button.BorderSizePixel = 0
		button.Parent = frame
		button.MouseButton1Click:Connect(callback)
	end

	makeButton("save", "Save Current", function()
		self:Save(workspace, "workspace")
	end)
	makeButton("autosave", "Start Auto Save", function()
		self:AutoSave(workspace, "workspace", self.options.autosaveInterval)
	end)
	makeButton("restore", "Restore Last Save", function()
		if self.lastPath then
			self:Restore(self.lastPath, workspace)
		end
	end)
	makeButton("backup", "Create Backup", function()
		self:Save(workspace, "backup", "backup")
	end)
	makeButton("stop", "Stop Auto Save", function()
		self:StopAutoSave()
	end)

	return screenGui
end

local saver = Saver.new({
	saveFolder = "InsaneSaver",
	backupFolder = "InsaneSaver/Backups",
	keepBackups = 12,
	autosaveInterval = 30,
	uiEnabled = true,
})

saver:CreateUI()

return saver
